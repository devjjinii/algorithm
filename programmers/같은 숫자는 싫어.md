## 같은 숫자는 싫어 (JavaScript)

* 내 풀이
~~~
// 중복제거
// 배열 하나 할당, 같은 숫자면 push안하고
// 다른 숫자면 push

function solution(arr)
{
    var answer = [];
    var temp = arr[0];
    answer.push(temp)
    // console.log("temp : ", temp)
    
    for(var i =1; i< arr.length; i++) {
        if(temp !== arr[i]) {
            temp = arr[i];
            // console.log("arr temp : ", temp)
            answer.push(temp);
        }
    }
    return answer;
}
~~~

* 다른 사람 풀이
~~~
function solution(arr)
{
    return arr.filter((val,index) => val != arr[index+1]);
}
